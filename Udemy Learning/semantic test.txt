        <article> -- self contain element independent/reusablehttps://developer.mozilla.org/en-US/docs/Web/HTML/Element/article
        <aside>-- not essential to the site or indirectlyrelated e.g the side barhttps://developer.mozilla.org/en-US/docs/Web/HTML/Element/aside
        <header> -- introductury content of site or of an article
        <footer> -- ending element of a site or an article
